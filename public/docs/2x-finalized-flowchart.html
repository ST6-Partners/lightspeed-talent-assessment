flowchart TD
  A["Job Posting"] --> B["Send in resume"]
  B --> C["Email — take assessment"]:::email
  C --> D["(1 week) Email reminder — take assessment"]:::email
  D --> E1["Doesn't complete assessment within 2 weeks of email"]
  D --> E2["Score < 30"]
  D --> E3["Score 30+"]
  E1 --> R1["Rejected"]:::reject
  R1 --> N1["Notified through email"]:::email
  E2 --> R2["Rejected"]:::reject
  R2 --> N2["Notified through email"]:::email
  E3 --> F["Notified through email + work sample"]:::email
  F --> G["Work Sample + resume + reference checks"]
  G --> R3["Not up to standard — rejected and notified by email"]:::reject
  G --> N3["Notified through email"]:::email
  N3 --> H["EPP vs. company values"]
  H --> R4["Values don't fit — rejected"]:::reject
  R4 --> N4["Notified through email"]:::email
  H --> I["Values fit"]
  I --> J["Email + schedule interview"]:::email
  J --> K["Interview with HR"]
  TQ["Results tailor interview questions"] --> K
  TQ --> TI["Email to interviewer"]:::email
  K --> L["Transcript processed — feedback for interviewer and candidate"]
  L --> M["Email feedback to interviewer"]:::email
  M --> R5["Rejected"]:::reject
  R5 --> N5["Notified through email"]:::email
  M --> O["Email feedback to candidate"]:::email
  O --> P["Bingo"]:::success
  P --> Q["Email + offer letter"]:::email
  Q --> Z["Insights"]
